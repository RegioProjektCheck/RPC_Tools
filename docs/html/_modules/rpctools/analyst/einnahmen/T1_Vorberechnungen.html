                ags_regenesis = ags_vg

    #Niedersachsen
    elif ags_input[0:2] == "03":

        if verbandsgemeinde[0:3] == "kre":
            ags_vg = ""
            ags_regenesis = ags_input

        elif verbandsgemeinde[0:3] == "gem":
            rows = arcpy.SearchCursor(VG250, where)
            for row in rows:
                ags_vg = RS[0:5] + RS[6] + RS[7:9]
                ags_regenesis = ags_vg

        else:
            ags_vg = ""
